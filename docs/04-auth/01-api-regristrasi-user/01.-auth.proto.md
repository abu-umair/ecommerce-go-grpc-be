```proto
syntax = "proto3"; 
// Versi syntax protobuf (wajib dideklarasikan)

package auth; 
// Sama seperti namespace Laravel: App\Http\Controllers\Auth

import "common/base_response.proto"; 
// Import untuk response standar (statusCode, message, dll) → mirip bikin Resource Laravel

import "buf/validate/validate.proto"; 
// Untuk validasi langsung di level protobuf → mirip FormRequest di Laravel

option go_package = "github.com/abu-umair/ecommerce-go-grpc-be/pb/auth"; 
// Hasil generate code Go akan masuk ke folder pb/auth
// Analogi: namespace otomatis saat bikin Controller di Laravel
  

// Sama kayak bikin AuthController dengan method register(Request $request)
service AuthService {
  rpc Register(RegisterRequest) returns (RegisterResponse);
  // Laravel: Route::post('/auth/register', [AuthController::class, 'register']);
}


// Request untuk registrasi user
message RegisterRequest { 
  // Laravel: FormRequest (aturan validasi input)
  
  string full_name = 1 [(buf.validate.field).string = { min_len: 1, max_len: 255 }];
  // Laravel: 'full_name' => 'required|string|max:255'

  string email = 2 [(buf.validate.field).string = { email: true, min_len: 1, max_len: 255 }];
  // Laravel: 'email' => 'required|email|max:255'

  string password = 3 [(buf.validate.field).string = { min_len: 1, max_len: 255 }];
  // Laravel: 'password' => 'required|string|max:255'

  string password_confirmation = 4 [(buf.validate.field).string = { min_len: 1, max_len: 255 }];
  // Laravel: 'password_confirmation' => 'required|same:password'
}


// Response dari registrasi user
message RegisterResponse { 
  common.BaseResponse base = 1; 
  // Laravel: return response()->json(['message' => 'User registered successfully'], 200);
}
```

## analogi Laravel
### routes/api.php
```php
// Definisi endpoint
Route::post('/auth/register', [AuthController::class, 'register']);

```

### FormRequest
```php
class RegisterRequest extends FormRequest {
    public function rules() {
        return [
            'full_name' => 'required|string|max:255',
            'email' => 'required|email|max:255',
            'password' => 'required|string|max:255',
            'password_confirmation' => 'required|same:password'
        ];
    }
}

```

### JsonResponse / Resource
```php
return response()->json([
    'statusCode' => 200,
    'message' => 'User registered successfully'
]);
```
